EndCombo() {
	LightCombo = faux
	MediumCombo = faux
	HeavyCombo = faux
}

Coroutine TempsMort() {
	Attendre x sec
}

LightATK() {
	SI (LightCombo=vrai&&MediumCombo=false&&HeavyCombo=false) {
		Lancer animation LightATK2
		SI (enemy touché) {
			Bléssé enemy
		} SINON {
			temps mort()
			EndCombo()
		}
	} SINON SI (LightCombo=false&&MediumCombo=false&&HeavyCombo=false) {
		Lancer animation LightATK1
		SI (enemy touché) {
			Bléssé enemy
			LightCombo = vrai
		} SINON {
			temps mort()
			EndCombo()
		}
	}
}

MediumATK() {
	SI ((LightCombo=vrai&&MediumCombo=vrai&&HeavyCombo=false)
		||(LightCombo=false&&MediumCombo=vrai&&HeavyCombo=false)) {
		Lancer animation MediumATK2
		SI (enemy touché) {
			Bléssé enemy
		} SINON {
			temps mort()
			EndCombo()
		}
	} SINON SI ((LightCombo=false&&MediumCombo=false&&HeavyCombo=false)
				||(LightCombo=vrai&&MediumCombo=false&&HeavyCombo=false)) {
		Lancer animation MediumATK1
		SI (enemy touché) {
			Bléssé enemy
			MediumCombo = vrai
		} SINON {
			temps mort()
			EndCombo()
		}
	}
}

HeavyATK() {
	SI ((LightCombo=false&&MediumCombo=false&&HeavyCombo=false)
		||(LightCombo=vrai&&MediumCombo=vrai&&HeavyCombo=false)
		||(LightCombo=false&&MediumCombo=vrai&&HeavyCombo=false)) {
		Lancer animation HeavyATK1
		SI (enemy touché) {
			Bléssé enemy
			HeavyCombo = vrai
		} SINON {
			temps mort()
			EndCombo()
		}
	}
}
